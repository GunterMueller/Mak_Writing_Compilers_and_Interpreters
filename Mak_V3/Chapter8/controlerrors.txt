    REPEAT UNTIL five := 5;
    REPEAT ratio := 9 UNTIL;

    WHILE DO five := 5;
    WHILE five = 5 five := 5 UNTIL ratio = 9;

    FOR i = 1 TO 10 DO five = 5;
    FOR i := 1, 10 DO five := 5;

    IF five = 5;
    IF five = 5 ELSE ratio := 9;
    IF five = 5 ELSE ratio := 9 THEN ratio := 7;

    CASE five OF
        1 2 3 : five := 5
        4,5     IF five = 5 THEN ratio := 9;
    END;
END.
